3use crate::diagnostics::diagnostics_api::{Diagnostic, Position, Range, Severity};
4use crate::parser::util::get_tree;
5use tree_sitter::Node;

7/// Detects very simple semantic issues in source and returns a vector
8/// of Diagnostics.
9pub fn detect_semantic_errors(source: &str) -> Vec<Diagnostic> {
10    let mut diagnostics = Vec::new();
11
12    let (tree, _source) = match get_tree(source) {
13        Some(tree) => tree,
14        // essence cannot be parsed
15        None => {
16            // get the position of last character to get the range of the entire source code
17            let last_line = source.lines().count().saturating_sub(1);
18            let last_char = source.lines().last().map(|l| l.len()).unwrap_or(0);
19            diagnostics.push(Diagnostic {
20                range: Range {
21                    start: Position {
22                        line: 0,
23                        character: 0,
24                    },
25                    end: Position {
26                        line: last_line as u32,
27                        character: last_char as u32,
28                    },
29                },
30                severity: Severity::Error,
31                message: "Failed to read the source code".to_string(),
32                source: "Tree-Sitter-Parse-Error",
33            });
34            return diagnostics;
35        }
36    };
37    let root_node = tree.root_node();
38
39    // print the tree
40    println!("=== Tree S-expression ===");
41    println!("{}", root_node.to_sexp());
42    println!("=========================\n");
43
44    // call semantic error detection functions
45    keyword_as_identifier(root_node, source, &mut diagnostics);
46
47    diagnostics
48}
49
50const KEYWORDS: [&str; 21] = [
51    "forall", "exists", "such", "that", "letting", "find", "minimise", "maximise", "subject", "to",
52    "where", "and", "or", "not", "if", "then", "else", "in", "sum", "product", "bool",
53];
54
55// keyword as identifier error and push to diagnostics
56fn keyword_as_identifier(root: Node, src: &str, diagnostics: &mut Vec<Diagnostic>) {
57    let mut stack = vec![root];
58    while let Some(node) = stack.pop() {
59        println!("Visiting node kind: {}", node.kind());
60        if (node.kind() == "variable" || node.kind() == "identifier" || node.kind() == "parameter")
61            && let Ok(text) = node.utf8_text(src.as_bytes())
62        {
63            let ident = text.trim();
64            if KEYWORDS.contains(&ident) {
65                let start_point = node.start_position();
66                let end_point = node.end_position();
67                diagnostics.push(Diagnostic {
68                    range: Range {
69                        start: Position {
70                            line: start_point.row as u32,
71                            character: start_point.column as u32,
72                        },
73                        end: Position {
74                            line: end_point.row as u32,
75                            character: end_point.column as u32,
76                        },
77                    },
78                    severity: Severity::Error,
79                    message: format!("Keyword '{}' used as an identifier", ident),
80                    source: "semantic-error-detector",
81                });
82                print!("{:?}", diagnostics.last().unwrap());
83            }
84        }
85
86        // push children onto stack (must be inside the while loop!)
87        for i in 0..node.child_count() {
88            if let Some(child) = node.child(i) {
89                stack.push(child);
90            }
91        }
92    }
