7/// Parse an Essence variable domain into its Conjure AST representation.
8pub fn parse_domain(domain: Node, source_code: &str) -> Result<Domain, EssenceParseError> {
9    match domain.kind() {
10        "domain" => parse_domain(domain.child(0).expect("No domain found"), source_code),
11        "bool_domain" => Ok(Domain::Bool),
12        "int_domain" => Ok(parse_int_domain(domain, source_code)),
13        "identifier" => {
14            let variable_name = &source_code[domain.start_byte()..domain.end_byte()];
15            Ok(Domain::Reference(Name::user(variable_name)))
16        }
17        "tuple_domain" => parse_tuple_domain(domain, source_code),
18        "matrix_domain" => parse_matrix_domain(domain, source_code),
19        "record_domain" => parse_record_domain(domain, source_code),
20        "set_domain" => parse_set_domain(domain, source_code),
21        _ => panic!("{} is not a supported domain type", domain.kind()),
22    }
23}

25/// Parse an integer domain. Can be a single integer or a range.
26fn parse_int_domain(int_domain: Node, source_code: &str) -> Domain {
27    if int_domain.child_count() == 1 {
28        Domain::Int(vec![Range::Bounded(i32::MIN, i32::MAX)])
29    } else {
30        let mut ranges: Vec<Range<i32>> = Vec::new();
31        let range_list = int_domain
32            .child_by_field_name("ranges")
33            .expect("No range list found (expression ranges not supported yet");
34        for int_range in named_children(&range_list) {
35            match int_range.kind() {
36                "integer" => {
37                    let integer_value = &source_code[int_range.start_byte()..int_range.end_byte()]
38                        .parse::<i32>()
39                        .unwrap();
40                    ranges.push(Range::Single(*integer_value));
41                }
42                "int_range" => {
43                    let lower_bound: Option<i32>;
44                    let upper_bound: Option<i32>;
45                    if let Some(lower_bound_node) = int_range.child_by_field_name("lower") {
46                        lower_bound = Some(
47                            source_code[lower_bound_node.start_byte()..lower_bound_node.end_byte()]
48                                .parse::<i32>()
49                                .unwrap(),
50                        );
51                    } else {
52                        lower_bound = None;
53                    }
54                    if let Some(upper_bound_node) = int_range.child_by_field_name("upper") {
55                        upper_bound = Some(
56                            source_code[upper_bound_node.start_byte()..upper_bound_node.end_byte()]
57                                .parse::<i32>()
58                                .unwrap(),
59                        );
60                    } else {
61                        upper_bound = None;
62                    }
63
64                    match (lower_bound, upper_bound) {
65                        (Some(lb), Some(ub)) => ranges.push(Range::Bounded(lb, ub)),
66                        (Some(lb), None) => ranges.push(Range::Bounded(lb, i32::MAX)),
67                        (None, Some(ub)) => ranges.push(Range::Bounded(i32::MIN, ub)),
68                        _ => panic!("Unsupported int range type"),
69                    }
70                }
71                _ => panic!("unsupported int range type"),
72            }
73        }
74        Domain::Int(ranges)
75    }
76}

78fn parse_tuple_domain(tuple_domain: Node, source_code: &str) -> Result<Domain, EssenceParseError> {
79    let mut domains: Vec<Domain> = Vec::new();
80    for domain in named_children(&tuple_domain) {
81        domains.push(parse_domain(domain, source_code)?);
82    }
83    Ok(Domain::Tuple(domains))
84}

86fn parse_matrix_domain(
87    matrix_domain: Node,
88    source_code: &str,
89) -> Result<Domain, EssenceParseError> {
90    let mut domains: Vec<Domain> = Vec::new();
91    let index_domain_list = matrix_domain
92        .child_by_field_name("index_domain_list")
93        .expect("No index domains found for matrix domain");
94    for domain in named_children(&index_domain_list) {
95        domains.push(parse_domain(domain, source_code)?);
96    }
97    let value_domain = parse_domain(
98        matrix_domain.child_by_field_name("value_domain").ok_or(
99            EssenceParseError::syntax_error(
100                "Expected a value domain".to_string(),
101                Some(matrix_domain.range()),
102            ),
103        )?,
104        source_code,
105    )?;
106    Ok(Domain::Matrix(Box::new(value_domain), domains))
107}

109fn parse_record_domain(
110    record_domain: Node,
111    source_code: &str,
112) -> Result<Domain, EssenceParseError> {
113    let mut record_entries: Vec<RecordEntry> = Vec::new();
114    for record_entry in named_children(&record_domain) {
115        let name_node = record_entry
116            .child_by_field_name("name")
117            .expect("No name found for record entry");
118        let name = Name::user(&source_code[name_node.start_byte()..name_node.end_byte()]);
119        let domain_node = record_entry
120            .child_by_field_name("domain")
121            .expect("No domain found for record entry");
122        let domain = parse_domain(domain_node, source_code)?;
123        record_entries.push(RecordEntry { name, domain });
124    }
125    Ok(Domain::Record(record_entries))
126}

128fn parse_set_domain(set_domain: Node, source_code: &str) -> Result<Domain, EssenceParseError> {
129    let mut set_attribute: Option<SetAttr> = None;
130    let mut value_domain: Option<Domain> = None;
131
132    for child in named_children(&set_domain) {
133        match child.kind() {
134            "set_attributes" => {
135                // Check if we have both minSize and maxSize (minMax case)
136                let min_value_node = child.child_by_field_name("min_value");
137                let max_value_node = child.child_by_field_name("max_value");
138                let size_value_node = child.child_by_field_name("size_value");
139
140                if let (Some(min_node), Some(max_node)) = (min_value_node, max_value_node) {
141                    // MinMax case
142                    let min_str = &source_code[min_node.start_byte()..min_node.end_byte()];
143                    let max_str = &source_code[max_node.start_byte()..max_node.end_byte()];
144
145                    let min_val = i32::from_str(min_str).map_err(|_| {
146                        EssenceParseError::syntax_error(
147                            format!("Invalid integer value for minSize: {}", min_str),
148                            Some(min_node.range()),
149                        )
150                    })?;
151
152                    let max_val = i32::from_str(max_str).map_err(|_| {
153                        EssenceParseError::syntax_error(
154                            format!("Invalid integer value for maxSize: {}", max_str),
155                            Some(max_node.range()),
156                        )
157                    })?;
158
159                    set_attribute = Some(SetAttr::MinMaxSize(min_val, max_val));
160                } else if let Some(size_node) = size_value_node {
161                    // Size case
162                    let size_str = &source_code[size_node.start_byte()..size_node.end_byte()];
163                    let size_val = i32::from_str(size_str).map_err(|_| {
164                        EssenceParseError::syntax_error(
165                            format!("Invalid integer value for size: {}", size_str),
166                            Some(size_node.range()),
167                        )
168                    })?;
169                    set_attribute = Some(SetAttr::Size(size_val));
170                } else if let Some(min_node) = min_value_node {
171                    // MinSize only case
172                    let min_str = &source_code[min_node.start_byte()..min_node.end_byte()];
173                    let min_val = i32::from_str(min_str).map_err(|_| {
174                        EssenceParseError::syntax_error(
175                            format!("Invalid integer value for minSize: {}", min_str),
176                            Some(min_node.range()),
177                        )
178                    })?;
179                    set_attribute = Some(SetAttr::MinSize(min_val));
180                } else if let Some(max_node) = max_value_node {
181                    // MaxSize only case
182                    let max_str = &source_code[max_node.start_byte()..max_node.end_byte()];
183                    let max_val = i32::from_str(max_str).map_err(|_| {
184                        EssenceParseError::syntax_error(
185                            format!("Invalid integer value for maxSize: {}", max_str),
186                            Some(max_node.range()),
187                        )
188                    })?;
189                    set_attribute = Some(SetAttr::MaxSize(max_val));
190                }
191            }
192            "domain" => {
193                value_domain = Some(parse_domain(child, source_code)?);
194            }
195            _ => {
196                return Err(EssenceParseError::syntax_error(
197                    format!("Unrecognized set domain child kind: {}", child.kind()),
198                    Some(child.range()),
199                ));
200            }
201        }
202    }
203
204    if let Some(domain) = value_domain {
205        Ok(Domain::Set(
206            set_attribute.unwrap_or(SetAttr::None),
207            Box::new(domain),
208        ))
209    } else {
210        Err(EssenceParseError::syntax_error(
211            "Set domain must have a value domain".to_string(),
212            Some(set_domain.range()),
213        ))
214    }
215}
