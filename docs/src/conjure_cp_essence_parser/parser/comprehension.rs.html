1use crate::expression::parse_expression;
2use crate::parser::domain::parse_domain;
3use crate::util::named_children;
4use crate::{EssenceParseError, field};
5use conjure_cp_core::ast::ac_operators::ACOperatorKind;
6use conjure_cp_core::ast::comprehension::ComprehensionBuilder;
7use conjure_cp_core::ast::{DeclarationPtr, Expression, Metadata, Moo, Name, SymbolTable};
8use std::cell::RefCell;
9use std::rc::Rc;
10use tree_sitter::Node;
11
12pub fn parse_comprehension(
13    node: &Node,
14    source_code: &str,
15    root: &Node,
16    symbols_ptr: Option<Rc<RefCell<SymbolTable>>>,
17) -> Result<Expression, EssenceParseError> {
18    // Comprehensions require a symbol table passed in
19    let symbols_ptr = symbols_ptr.ok_or_else(|| {
20        EssenceParseError::syntax_error(
21            "Comprehensions require a symbol table".to_string(),
22            Some(node.range()),
23        )
24    })?;
25
26    let mut builder = ComprehensionBuilder::new(symbols_ptr);
27
28    // We need to track the return expression node separately since it appears first in syntax
29    // but we need to parse generators first (to get variables in scope)
30    let mut return_expr_node: Option<Node> = None;
31
32    // set return expression node and parse generators/conditions
33    for child in named_children(node) {
34        match child.kind() {
35            "arithmetic_expr" | "bool_expr" | "comparison_expr" => {
36                // Store the return expression node to parse later
37                return_expr_node = Some(child);
38            }
39            "generator" => {
40                // Parse the generator variable
41                let var_node = field!(child, "variable");
42                let var_name_str = &source_code[var_node.start_byte()..var_node.end_byte()];
43                let var_name = Name::user(var_name_str);
44
45                // Parse the domain
46                let domain_node = field!(child, "domain");
47                let var_domain = parse_domain(domain_node, source_code)?;
48
49                // Add generator using the builder
50                let decl = DeclarationPtr::new_var(var_name, var_domain);
51                builder = builder.generator(decl);
52            }
53            "condition" => {
54                // Parse the condition expression
55                let expr_node = field!(child, "expression");
56                let generator_symboltable = builder.generator_symboltable();
57
58                let guard_expr =
59                    parse_expression(expr_node, source_code, root, Some(generator_symboltable))?;
60
61                // Add the condition as a guard
62                builder = builder.guard(guard_expr);
63            }
64            _ => {
65                // Skip other nodes (like punctuation)
66            }
67        }
68    }
69
70    // parse the return expression
71    let return_expr_node = return_expr_node.ok_or_else(|| {
72        EssenceParseError::syntax_error(
73            "Comprehension missing return expression".to_string(),
74            Some(node.range()),
75        )
76    })?;
77
78    // Use the return expression symbol table which already has induction variables (as Given) and parent as parent
79    let return_expr = parse_expression(
80        return_expr_node,
81        source_code,
82        root,
83        Some(builder.return_expr_symboltable()),
84    )?;
85
86    // Build the comprehension with the return expression and default ACOperatorKind::And
87    let comprehension = builder.with_return_value(return_expr, Some(ACOperatorKind::And));
88
89    Ok(Expression::Comprehension(
90        Metadata::new(),
91        Moo::new(comprehension),
92    ))
93}
