12/// Maps CO variable names to their CO domains, Z3 symbolic constants, and Z3 symbols.
13#[derive(Clone, Debug)]
14pub struct SymbolStore {
15    map: HashMap<Name, (Domain, Dynamic, Symbol)>,
16    theories: TheoryConfig,
17}
18
19impl SymbolStore {
20    pub fn new(theories: TheoryConfig) -> Self {
21        SymbolStore {
22            map: HashMap::new(),
23            theories,
24        }
25    }
26
27    pub fn insert(
28        &mut self,
29        name: Name,
30        val: (Domain, Dynamic, Symbol),
31    ) -> Option<(Domain, Dynamic, Symbol)> {
32        self.map.insert(name, val)
33    }
34
35    pub fn get(&self, name: &Name) -> Option<&(Domain, Dynamic, Symbol)> {
36        self.map.get(name)
37    }
38}

40impl Solvable for SymbolStore {
41    type ModelInstance = LiteralStore;
42
43    fn read_from_model(
44        &self,
45        model: &Model,
46        model_completion: bool,
47    ) -> Option<Self::ModelInstance> {
48        let mut new_store = LiteralStore::new();
49        for (name, (domain, ast, sym)) in self.map.iter() {
50            // Get the interpretation of each constant
51            let (ast, lit) =
52                interpret(model, (domain, ast), model_completion, &self.theories).unwrap();
53            new_store.map.insert(name.clone(), (ast, lit));
54        }
55        Some(new_store)
56    }
57
58    fn generate_constraint(&self, model: &Self::ModelInstance) -> Bool {
59        let bools: Vec<_> = self
60            .map
61            .iter()
62            .map(|(name, (domain, ast, _))| {
63                let (other, _) = model.map.get(name).unwrap();
64                match domain {
65                    Domain::Matrix(_, idx_domains) => {
66                        // Rather than just setting `array != other_array`, we need to do it for every element
67                        // Otherwise, Z3 can generate new arrays which are equal over the domain
68                        // but different otherwise (and this loops infinitely)
69
70                        let neqs: Vec<_> = idx_domains
71                            .iter()
72                            .map(|domain| domain_to_ast_vec(&self.theories, domain).unwrap())
73                            .multi_cartesian_product()
74                            .map(|idxs| {
75                                idxs.iter()
76                                    .fold((ast.clone(), other.clone()), |(a, b), idx| {
77                                        (
78                                            a.as_array().unwrap().select(idx),
79                                            b.as_array().unwrap().select(idx),
80                                        )
81                                    })
82                            })
83                            .map(|(a, b)| a.ne(b))
84                            .collect();
85                        Bool::or(&neqs)
86                    }
87
88                    // Any other variables are just directly compared
89                    _ => ast.ne(other),
90                }
91            })
92            .collect();
93        Bool::or(bools.as_slice())
94    }
95}

97/// Maps CO variable names to their literal values in both the Z3 model and Essence.
98#[derive(Clone, Debug)]
99pub struct LiteralStore {
100    map: HashMap<Name, (Dynamic, Literal)>,
101}
102
103impl LiteralStore {
104    fn new() -> Self {
105        LiteralStore {
106            map: HashMap::new(),
107        }
108    }
109
110    /// Return this store as a mapping of CO names to literals
111    pub fn as_literals_map(&self) -> SolverResult<HashMap<Name, Literal>> {
112        let iter = self
113            .map
114            .iter()
115            .map(|(name, (_, lit))| (name.clone(), lit.clone()));
116        Ok(HashMap::from_iter(iter))
117    }
118}

135/// Interprets the given value within the given model and returns a CO literal.
136/// The value can be any AST including a symbolic constant, as long as it is defined in the model.
137///
138/// This method makes any interpretations necessary to fully evaluate the literal. E.g. for arrays
139/// it must enumerate over all elements in the index domain and evaluate the elements as literals.
140fn interpret(
141    model: &Model,
142    value: (&Domain, &Dynamic),
143    model_completion: bool,
144    theories: &TheoryConfig,
145) -> SolverResult<(Dynamic, Literal)> {
146    use IntTheory::{Bv, Lia};
147
148    let (domain, var_ast) = value;
149    let lit_ast = model
150        .eval(var_ast, model_completion)
151        .ok_or(SolverError::Runtime(format!(
152            "could not interpret variable: {var_ast}"
153        )))?;
154
155    let literal = match (theories.ints, lit_ast.sort_kind()) {
156        (_, SortKind::Bool) => {
157            let bool_ast = lit_ast.as_bool().unwrap();
158            let bool = bool_ast.as_bool().unwrap();
159            Ok(Literal::Bool(bool))
160        }
161        (Lia, SortKind::Int) => {
162            let int_ast = lit_ast.as_int().unwrap();
163            let int = int_ast
164                .as_i64()
165                .ok_or(SolverError::Runtime(format!(
166                    "could not cast to i64: {lit_ast}"
167                )))?
168                .try_into()
169                .map_err(|err| {
170                    SolverError::Runtime(format!("value {lit_ast} out of range: {err}"))
171                })?;
172            Ok(Literal::Int(int))
173        }
174        (Bv, SortKind::BV) => {
175            // BVs do not sign-extend when returning u64s (if they are < 64 bits)
176            // To correctly retrieve negative numbers, we downsize to a u32 and then bit-wise
177            // interpret it as an i32, rather than casting.
178            // See https://github.com/prove-rs/z3.rs/issues/458
179            let bv_ast = lit_ast.as_bv().unwrap();
180            let unsigned_64: u64 = bv_ast.as_u64().ok_or(SolverError::Runtime(format!(
181                "could not retrieve u64: {lit_ast}"
182            )))?;
183            let unsigned_32: u32 = unsigned_64.try_into().map_err(|err| {
184                SolverError::Runtime(format!("value {lit_ast} out of range: {err}"))
185            })?;
186            let signed = i32::from_ne_bytes(unsigned_32.to_ne_bytes());
187            Ok(Literal::Int(signed))
188        }
189        (_, SortKind::Array) => {
190            let arr_ast = lit_ast.as_array().unwrap();
191            let Domain::Matrix(val_domain, idx_domains) = domain else {
192                return Err(SolverError::Runtime(format!(
193                    "non-matrix variable interpreted as array: {domain}"
194                )));
195            };
196
197            let inner_domain = match idx_domains.as_slice() {
198                [idx_domain] => *val_domain.clone(),
199                [idx_domain, tail @ ..] => Domain::Matrix(val_domain.clone(), tail.to_vec()),
200                [] => return Err(SolverError::Runtime("empty matrix index domain".into())),
201            };
202
203            let indices = domain_to_ast_vec(theories, &idx_domains[0])?;
204            let elements_res: Result<Vec<_>, _> = indices
205                .iter()
206                .map(|idx| model.eval(&arr_ast.select(idx), model_completion).unwrap())
207                .map(|ast| interpret(model, (&inner_domain, &ast), model_completion, theories))
208                .map(|res| res.map(|(_, lit)| lit))
209                .collect();
210            let elements = elements_res?;
211
212            Ok(Literal::AbstractLiteral(AbstractLiteral::Matrix(
213                elements,
214                Box::new(domain.clone()),
215            )))
216        }
217        _ => Err(SolverError::RuntimeNotImplemented(format!(
218            "conversion from AST to literal not implemented: {lit_ast}"
219        ))),
220    }?;
221    Ok((lit_ast, literal))
222}
