9/// A [SolverAdaptor] for interacting with SMT solvers, specifically Z3.
10pub struct Smt {
11    __non_constructable: private::Internal,
12
13    /// Initially maps variables to unknown constants.
14    /// Also used to store their solved literal values.
15    store: SymbolStore,
16
17    /// Assertions are added to this solver instance when loading the model.
18    solver_inst: Solver,

20    theory_config: TheoryConfig,
21}

25impl Default for Smt {

36impl Smt {
37    /// Constructs a new adaptor using the given theories for representing the relevant constructs.
38    pub fn new(int_theory: IntTheory, matrix_theory: MatrixTheory) -> Self {
39        let theories = TheoryConfig {
40            ints: int_theory,
41            matrices: matrix_theory,
42        };
43        Smt {
44            theory_config: theories.clone(),
45            store: SymbolStore::new(theories),
46            ..Default::default()
47        }
48    }
49}

51impl SolverAdaptor for Smt {
52    fn solve(
53        &mut self,
54        callback: SolverCallback,
55        _: private::Internal,
56    ) -> Result<SolveSuccess, SolverError> {
57        let solutions = self
58            .solver_inst
59            .solutions(&self.store, true)
60            .take_while(|store| (callback)(store.as_literals_map().unwrap()));
61
62        // Consume iterator and get whether there are solutions
63        let search_complete = match solutions.count() {
64            0 => SearchComplete::NoSolutions,
65            _ => SearchComplete::HasSolutions,
66        };
67
68        Ok(SolveSuccess {
69            // TODO: get solver stats
70            stats: Default::default(),
71            status: SearchStatus::Complete(search_complete),
72        })
73    }
74
75    fn solve_mut(
76        &mut self,
77        callback: SolverMutCallback,
78        _: private::Internal,
79    ) -> Result<SolveSuccess, SolverError> {
80        Err(SolverError::OpNotImplemented("solve_mut".into()))
81    }
82
83    fn load_model(&mut self, model: Model, _: private::Internal) -> Result<(), SolverError> {
84        let submodel = model.as_submodel();
85        load_model_impl(
86            &mut self.store,
87            &mut self.solver_inst,
88            &self.theory_config,
89            &submodel.symbols(),
90            submodel.constraints().as_slice(),
91        )?;
92        Ok(())
93    }
94
95    fn get_family(&self) -> SolverFamily {
96        SolverFamily::Smt
97    }
98
99    fn get_name(&self) -> Option<String> {
100        Some("SMT".to_string())
101    }
102
103    fn write_solver_input_file(
104        &self,
105        writer: &mut impl std::io::Write,
106    ) -> Result<(), std::io::Error> {
107        let smt2 = self.solver_inst.to_smt2();
108        writer.write(smt2.as_bytes()).map(|_| ())
109    }
