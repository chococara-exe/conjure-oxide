9/// Use 32-bit 2's complement signed bit-vectors
10pub const BV_SIZE: u32 = 32;
11
12/// A function which encodes a restriction for a specific variable. Given an AST of the correct
13/// sort, constructs a boolean assertion which will ensure the variable has the correct domain.
14type RestrictFn = Box<dyn Fn(&Dynamic) -> Bool>;
15
16/// Returns the Oxide domain as a Z3 sort, along with a function to restrict a variable of that sort
17/// to the original domain's restrictions.
18pub fn domain_to_sort(
19    domain: &Domain,
20    theories: &TheoryConfig,
21) -> SolverResult<(Sort, RestrictFn)> {
22    use IntTheory::{Bv, Lia};
23
24    match (theories.ints, domain) {
25        // Booleans of course have the same domain in SMT, so no restriction required
26        (_, Domain::Bool) => Ok((Sort::bool(), Box::new(|_| Bool::from_bool(true)))),
27
28        // Return a disjunction of the restrictions each range of the domain enforces
29        // I.e. `x: int(1, 3..5)` -> `or([x = 1, x >= 3 /\ x <= 5])`
30        (Lia, Domain::Int(ranges)) => {
31            let ranges = ranges.clone();
32            let restrict_fn = move |ast: &Dynamic| {
33                let int = ast.as_int().unwrap();
34                let restrictions: Vec<_> = ranges
35                    .iter()
36                    .map(|range| int_range_to_int_restriction(&int, range))
37                    .collect();
38                Bool::or(restrictions.as_slice())
39            };
40            Ok((Sort::int(), Box::new(restrict_fn)))
41        }
42        (Bv, Domain::Int(ranges)) => {
43            let ranges = ranges.clone();
44            let restrict_fn = move |ast: &Dynamic| {
45                let bv = ast.as_bv().unwrap();
46                let restrictions: Vec<_> = ranges
47                    .iter()
48                    .map(|range| int_range_to_bv_restriction(&bv, range))
49                    .collect();
50                Bool::or(restrictions.as_slice())
51            };
52            Ok((Sort::bitvector(BV_SIZE), Box::new(restrict_fn)))
53        }
54
55        (_, Domain::Matrix(val_domain, idx_domains)) => {
56            // We constrain the inner values of the domain recursively
57            // I.e. every way to index the array must give a value in the correct domain
58
59            let (range_sort, restrict_val) = match idx_domains.as_slice() {
60                [_] => domain_to_sort(val_domain, theories),
61                [_, tail @ ..] => {
62                    // Treat as a matrix containing (n-1)-dimensional matrices
63                    let inner_domain = Domain::Matrix(val_domain.clone(), tail.to_vec());
64                    domain_to_sort(&inner_domain, theories)
65                }
66                [] => Err(SolverError::ModelInvalid(
67                    "empty matrix index domain".into(),
68                )),
69            }?;
70            let idx_domain = &idx_domains[0];
71            let (domain_sort, _) = domain_to_sort(idx_domain, theories)?;
72
73            let idx_asts = domain_to_ast_vec(theories, idx_domain)?;
74            let restrict_fn = move |ast: &Dynamic| {
75                let arr = ast.as_array().unwrap();
76                let restrictions: Vec<_> = idx_asts
77                    .iter()
78                    .map(|idx_ast| (restrict_val)(&arr.select(idx_ast)))
79                    .collect();
80                Bool::and(restrictions.as_slice())
81            };
82            Ok((
83                Sort::array(&domain_sort, &range_sort),
84                Box::new(restrict_fn),
85            ))
86        }
87
88        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
89            "sort for '{domain}' not implemented"
90        ))),
91    }
92}

94/// Returns a domain as a vector of Z3 AST literals.
95pub fn domain_to_ast_vec(
96    theory_config: &TheoryConfig,
97    domain: &Domain,
98) -> SolverResult<Vec<Dynamic>> {
99    let lits = domain
100        .values()
101        .map_err(|err| SolverError::Runtime(err.to_string()))?;
102    lits.iter()
103        .map(|lit| literal_to_ast(theory_config, lit))
104        .collect()
105}

107/// Returns a boolean expression restricting the given integer variable to the given range.
108pub fn int_range_to_int_restriction(var: &Int, range: &Range<i32>) -> Bool {
109    match range {
110        Range::Single(n) => var.eq(Int::from(*n)),
111        Range::UnboundedL(r) => var.le(Int::from(*r)),
112        Range::UnboundedR(l) => var.ge(Int::from(*l)),
113        Range::Bounded(l, r) => Bool::and(&[var.ge(Int::from(*l)), var.le(Int::from(*r))]),
114    }
115}
116
117/// Returns a boolean expression restricting the given bitvector variable to the given integer range.
118pub fn int_range_to_bv_restriction(var: &BV, range: &Range<i32>) -> Bool {
119    match range {
120        Range::Single(n) => var.eq(BV::from_i64(*n as i64, BV_SIZE)),
121        Range::UnboundedL(r) => var.bvsle(BV::from_i64(*r as i64, BV_SIZE)),
122        Range::UnboundedR(l) => var.bvsge(BV::from_i64(*l as i64, BV_SIZE)),
123        Range::Bounded(l, r) => Bool::and(&[
124            var.bvsge(BV::from_i64(*l as i64, BV_SIZE)),
125            var.bvsle(BV::from_i64(*r as i64, BV_SIZE)),
126        ]),
127    }
128}

130pub fn name_to_symbol(name: &Name) -> SolverResult<Symbol> {
131    match name {
132        Name::User(ustr) => Ok(Symbol::String((*ustr).into())),
133        Name::Machine(num) => Ok(Symbol::Int(*num as u32)),
134        Name::Represented(parts) => {
135            let (name, rule_str, suffix) = parts.as_ref();
136            Ok(Symbol::String(format!("{name}#{rule_str}_{suffix}")))
137        }
138        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
139            "variable '{name}' name is unsupported"
140        ))),
141    }
142}

144/// Converts an atom (literal or reference) into an AST node.
145pub fn atom_to_ast(
146    theory_config: &TheoryConfig,
147    store: &SymbolStore,
148    atom: &Atom,
149) -> SolverResult<Dynamic> {
150    match atom {
151        Atom::Reference(decl) => store
152            .get(&decl.name())
153            .ok_or(SolverError::ModelInvalid(format!(
154                "variable '{}' does not exist",
155                decl.name()
156            )))
157            .map(|(_, ast, _)| ast)
158            .cloned(),
159        Atom::Literal(lit) => literal_to_ast(theory_config, lit),
160        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
161            "atom sort not implemented: {atom}"
162        ))),
163    }
164}
165
166/// Converts a CO literal (expression containing no variables) into an AST node.
167pub fn literal_to_ast(theory_config: &TheoryConfig, lit: &Literal) -> SolverResult<Dynamic> {
168    match lit {
169        Literal::Bool(b) => Ok(Bool::from_bool(*b).into()),
170        Literal::Int(n) => Ok(match theory_config.ints {
171            IntTheory::Lia => Int::from(*n).into(),
172            IntTheory::Bv => BV::from_i64(*n as i64, BV_SIZE).into(),
173        }),
174        _ => Err(SolverError::ModelFeatureNotImplemented(format!(
175            "literal type not implemented: {lit}"
176        ))),
177    }
